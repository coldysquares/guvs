<script>

    // -------------------------------------------------------------------------

    // 1. DATA SEEDING (Protocellular Coacervate Compartments & Hyphal Flux Paths)

    // -------------------------------------------------------------------------

    const compartments = [

      { id: 0, name: "Source-Compiler-Core", type: "Fluid-Dense", tokens: "42,500", volume: "0.425", valency: "7.842", entropy: "1.214", desc: "Houses core architecture scripts, abstract syntax tree rules, and python layer mappings." },

      { id: 1, name: "Regulatory-Compliance-Tax", type: "Fluid-Isolated", tokens: "15,000", volume: "0.150", valency: "2.104", entropy: "4.812", desc: "Corporate auditing profiles, structural compliance layers, and localized legal texts." },

      { id: 2, name: "Mathematical-Proof-Lib", type: "Ostwald-Ripened", tokens: "6,700", volume: "0.067", valency: "6.118", entropy: "0.925", desc: "Fused semantic equations, complex Eulerian amplitudes, and matrix validation logic arrays." },

      { id: 3, name: "Temporal-Context-Droplet", type: "Porous-Nucleating", tokens: "1,200", volume: "0.012", valency: "4.551", entropy: "2.314", desc: "Volatile conversational inputs and immediate environmental runtime adjustments." }

    ];

    // 1.27 Mycelian Hyphal Interconnectivity Matrix (Flux coefficients)

    const hyphalLinks = [

      { source: 0, target: 2, flux: "0.941", type: "high" },

      { source: 0, target: 3, flux: "0.684", type: "mid" },

      { source: 2, target: 3, flux: "0.412", type: "low" },

      { source: 1, target: 3, flux: "0.104", type: "low" }

    ];

    // -------------------------------------------------------------------------

    // 2. SIMULATION & SPRING PHYSICS RENDER ENGINE

    // -------------------------------------------------------------------------

    const canvas = document.getElementById("constellation");

    const ctx = canvas.getContext("2d");

   

    let width = canvas.width = canvas.offsetWidth;

    let height = canvas.height = canvas.offsetHeight;

    // Node spatial instantiation configurations

    const nodes = compartments.map((c, i) => ({

      ...c,

      x: width / 2 + (Math.cos(i * 1.5) * 120),

      y: height / 2 + (Math.sin(i * 1.5) * 120),

      vx: 0,

      vy: 0,

      radius: parseFloat(c.volume) * 40 + 20

    }));

    let selectedNodeId = null;

    let dragNode = null;

    let panX = 0, panY = 0;

    let isPanning = false;

    let startX, startY;

    // Window adjustment tracker

    window.addEventListener("resize", () => {

      width = canvas.width = canvas.offsetWidth;

      height = canvas.height = canvas.offsetHeight;

    });

    // -------------------------------------------------------------------------

    // 3. EVENT ROUTING & GRAPH SELECTION INTERACTION INTERFACE

    // -------------------------------------------------------------------------

    canvas.addEventListener("mousedown", (e) => {

      const rect = canvas.getBoundingClientRect();

      const mouseX = e.clientX - rect.left - panX;

      const mouseY = e.clientY - rect.top - panY;

      // Intersection Check: Did the operator click a protocellular node?

      dragNode = nodes.find(n => {

        const dx = n.x - mouseX;

        const dy = n.y - mouseY;

        return Math.sqrt(dx * dx + dy * dy) < n.radius;

      });

      if (dragNode) {

        selectedNodeId = dragNode.id;

        updateInspector(dragNode.id);

        canvas.classList.add("dragging");

      } else {

        isPanning = true;

        startX = e.clientX - panX;

        startY = e.clientY - panY;

      }

    });

    canvas.addEventListener("mousemove", (e) => {

      if (dragNode) {

        const rect = canvas.getBoundingClientRect();

        dragNode.x = e.clientX - rect.left - panX;

        dragNode.y = e.clientY - rect.top - panY;

      } else if (isPanning) {

        panX = e.clientX - startX;

        panY = e.clientY - startY;

      }

    });

    canvas.addEventListener("mouseup", () => {

      dragNode = null;

      isPanning = false;

      canvas.classList.remove("dragging");

    });

    // Tap-to-traverse connector action function

    window.navigateToNode = function(id) {

      selectedNodeId = id;

      updateInspector(id);

    };

    function updateInspector(id) {

      const node = nodes.find(n => n.id === id);

      const content = document.getElementById("inspector-content");

     

      // Filter the 1.27 Mycelian connections corresponding to this segment

      const activeLinks = hyphalLinks.filter(l => l.source === id || l.target === id);

     

      let linksHTML = activeLinks.map(l => {

        const targetId = l.source === id ? l.target : l.source;

        const targetNode = nodes.find(n => n.id === targetId);

        let colorClass = `var(--flux-${l.type})`;

        return `

          <div class="hyphal-link" onclick="navigateToNode(${targetId})">

            <div class="dot" style="background: ${colorClass}"></div>

            <span>${targetNode.name}</span>

            <small>Ψ: ${l.flux}</small>

          </div>

        `;

      }).join("");

      content.innerHTML = `

        <div class="eyebrow">Active Compartment Profile</div>

        <h2 class="compartment-name">${node.name}</h2>

        <div class="compartment-status">// Phase State: ${node.type}</div>

       

        <div class="field"><b>Token Volume Fraction (Φ)</b><span>${node.volume} (${node.tokens} tokens)</span></div>

        <div class="field"><b>Structural Valency (V_p)</b><span>${node.valency} active heads</span></div>

        <div class="field"><b>Localized Functional Entropy</b><span>${node.entropy} nats</span></div>

        <div class="field"><b>Droplet Matrix Substrate Profile</b><div style="font-size:12px; color:var(--muted); margin-top:4px;">${node.desc}</div></div>

        <div class="eyebrow" style="margin-top:20px;">Hyphal Links (+27% Expansion)</div>

        <div class="hyphal-conns">${linksHTML || '<div class="empty">No active structural hypha links.</div>'}</div>

      `;

    }

    // -------------------------------------------------------------------------

    // 4. ANIMATION TICK & LAYER FUSION DRAW ENGINE

    // -------------------------------------------------------------------------

    function step() {

      ctx.clearRect(0, 0, width, height);

      ctx.save();

      ctx.translate(panX, panY);

      // Physics Loop: Apply network spring constraints and node damping variables

      hyphalLinks.forEach(link => {

        const s = nodes[link.source];

        const t = nodes[link.target];

        const dx = t.x - s.x;

        const dy = t.y - s.y;

        const dist = Math.sqrt(dx * dx + dy * dy) || 1;

        const targetDist = 180;

       

        // 1.27 Mycelian interconnectivity factor adjustments applied to edge tension calculation

        const kSpring = 0.015 * parseFloat(link.flux) * 1.27;

        const force = (dist - targetDist) * kSpring;

       

        const fx = (dx / dist) * force;

        const fy = (dy / dist) * force;

        if (s !== dragNode) { s.vx += fx; s.vy += fy; }

        if (t !== dragNode) { t.vx -= fx; t.vy -= fy; }

      });

      // Compute centralized push forces to secure components cleanly on the grid screen

      nodes.forEach(n => {

        if (n !== dragNode) {

          n.x += n.vx;

          n.y += n.vy;

          n.vx *= 0.85;

          n.vy *= 0.85;

        }

      });

      // Render Structural Channels: Drawing the mycelian tracking pathways

      hyphalLinks.forEach(link => {

        const s = nodes[link.source];

        const t = nodes[link.target];

        ctx.beginPath();

        ctx.moveTo(s.x, s.y);

        ctx.lineTo(t.x, t.y);

       

        let strokeColor = "rgba(124, 92, 255, 0.2)";

        if (link.type === 'high') strokeColor = "rgba(126, 228, 210, 0.45)";

        if (link.type === 'mid') strokeColor = "rgba(124, 92, 255, 0.35)";

       

        ctx.strokeStyle = strokeColor;

        ctx.lineWidth = parseFloat(link.flux) * 3 * 1.27; // Augmented via the 1.27 scaling coefficient

        ctx.stroke();

      });

      // Render Coacervates: Drawing the porous protocellular lipid boundary rings

      nodes.forEach(n => {

        // Draw porous outer diffuse matrix layer rings

        ctx.beginPath();

        ctx.arc(n.x, n.y, n.radius + 6, 0, Math.pi * 2);

        ctx.strokeStyle = (n.id === selectedNodeId) ? "rgba(126, 228, 210, 0.8)" : "rgba(108, 133, 204, 0.15)";

        ctx.lineWidth = (n.id === selectedNodeId) ? 2 : 1;

        ctx.stroke();

        // Core coacervate particle fill representation

        ctx.beginPath();

        ctx.arc(n.x, n.y, n.radius, 0, Math.pi * 2);

       

        let gradient = ctx.createRadialGradient(n.x, n.y, 2, n.x, n.y, n.radius);

        gradient.addColorStop(0, 'rgba(14, 25, 48, 0.95)');

        gradient.addColorStop(1, 'rgba(9, 13, 24, 0.9)');

        ctx.fillStyle = gradient;

        ctx.fill();

       

        ctx.strokeStyle = (n.id === selectedNodeId) ? "var(--teal)" : "rgba(124, 92, 255, 0.4)";

        ctx.lineWidth = (n.id === selectedNodeId) ? 2 : 1.5;

        ctx.stroke();

        // Structural label details layer inside the nodes

        ctx.fillStyle = "var(--ink)";

        ctx.font = "11px monospace";

        ctx.textAlign = "center";

        ctx.fillText(n.name, n.x, n.y - 2);

       

        ctx.fillStyle = "var(--gold)";

        ctx.font = "9px -apple-system";

        ctx.fillText(`V_p: ${n.valency}`, n.x, n.y + 12);

      });

      ctx.restore();

      requestAnimationFrame(step);

    }

    // Initialize System Loop Runtime Pipeline execution

    requestAnimationFrame(step);

  </script>

</body>

</html>
